\startEXERCISE
給出迭代版的 \ALGO{RANDOMIZED-SELECT} （無遞迴調用）。
\stopEXERCISE

\startANSWER
\CLRSH{RANDOMIZED-SELECT-LOOP(A, p, r, i)}
\startCLRSCODE
while p < r
	q = \ALGO{RANDOMIZED-PARTITION(A,p,r)}
	k = q - p + 1
	if i == k
		return A[q]
	elseif i < k
		r = q - 1
	else
		p = q + 1
		i = i - k
return A[p]
\stopCLRSCODE
\stopANSWER
