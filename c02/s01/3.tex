\startCLRSCODE
for j = 2 to A.length
	key = A[j]
	// 將 $A[j]$ 插入到已排序的 $A[1 .. j-1]$ 中。
	i = j - 1
	while i > 0 and A[i] > key
		A[i+1] = A[i]
		i = i - 1
	A[i + 1] = key
\stopCLRSCODE
