\CLRSH{COUNTING-SORT(A,n,k)}
\startCLRSCODE
\ALGO{NEW-ARRAY(B[1:n])} and \ALGO{NEW-ARRAY(C[0:k])}
for i = 0 to k
	C[i] = 0
for j = 1 to n
	C[A[j]] = C[A[j]] + 1
// 現在 $C[i]$ 即爲 $i$ 在 $A$ 中出現的次數
for i = 1 to k
	C[i] = C[i] + C[i-1]
// C[i] 爲 $\le i$ 的元素的數目
// 將 $A$ 複製到 $B$ 中，從 $A$ 的末尾開始
for j = n downto 1
	B[C[A[j]]] = A[j]
	C[A[j]] = C[A[j]] - 1	// 處理重複的值
\stopCLRSCODE
